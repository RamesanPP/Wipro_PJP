DECLARE 
EMPNO NUMBER :=&empno;

BEGIN
IF NOT EXISTS(EMPNO) THEN
	INSERT INTO EMP VALUES EMPNO,ENAME,SAL WHERE ENAME=&ENAME, SAL=&SAL
ELSE
	UPDATE EMP SET ENAME=&ENAME, SAL=&SAL WHERE EMPNO
ENDIF;
END;
/

DECLARE 
NUM NUMBER :=&NUM;

BEGIN
IF NUM % 2 == 0 THEN
	dbms_output.put_line('Even')
ELSE
	dbms_output.put_line('Odd')
ENDIF;
END;
/

DECLARE 
EMPNO NUMBER :=&empno;

BEGIN
IF (SELECT DEPTNO FROM EMP WHERE EMPNO) = 10 THEN
	UPDATE EMP SET SAL=SAL+SAL*0.1 WHERE EMPNO
ELSIF (SELECT DEPTNO FROM EMP WHERE EMPNO) = 20 THEN
	UPDATE EMP SET SAL=SAL+SAL*0.15 WHERE EMPNO
ELSE
	UPDATE EMP SET SAL=SAL+COMM WHERE EMPNO
ENDIF;
END;
/
